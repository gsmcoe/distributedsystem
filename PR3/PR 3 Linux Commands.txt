STEP 4: paste the code 
#include <stdio.h>
#include <omp.h>
intmain()
{
int n, i;	int sum = 0;
printf("Enter number of elements: ");	scanf("%d", &n);
intarr[n];
printf("Enter elements:\n");
    for(i = 0; i< n; i++) 
{
scanf("%d", &arr[i]);
    }
    #pragma omp parallel num_threads(2)
    {
intlocal_sum = 0;
intthread_id = omp_get_thread_num();
inttotal_threads = omp_get_num_threads();
        for(i = thread_id; i< n; i += total_threads) 
	{
local_sum += arr[i];
        }

printf("Thread %d partial sum = %d\n", thread_id, local_sum);

        #pragma omp atomic
        sum += local_sum;
    }
printf("Final Sum = %d\n", sum);
    return 0;
}
/*OUTPUT: Enter number of elements: 4
Enter elements:1 2 3 4
Output:Thread 0 partial sum = 4
Thread 1 partial sum = 6
Final Sum = 10
*/
